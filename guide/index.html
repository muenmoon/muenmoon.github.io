    <template>
  <div>
    <AutoFilters
      :config="config"
      @handleSubmit="handleSubmit"
      :params="params"
      :gridCol="{ span: 8 }"
    >
    </AutoFilters>
  </div>
</template>

<script>
  export default {
    components: {},
    name: "HelloWorld",
    props: {},
    data() {
      return {
        regionList: [],
        params: {
          name: "",
          name1: "",
          sex: "",
          cascader: "",
          datePicker: "",
          radio: "",
          checkbox: [],
          autocomplete: "",
          region: [],
        },
      };
    },
    computed: {
      config() {
        return [
          {
            element: "el-input",
            field: "name",
            title: "名字",
            slots: {
              suffix: <i class="el-input__icon el-icon-date"></i>,
            },
          },
          {
            element: "el-autocomplete",
            field: "autocomplete",
            title: "输入建议 ",
            props: {
              fetchSuggestions: async (queryString, cb) => {
                const res = await new Promise((resolve) => {
                  setTimeout(() => {
                    const restaurant = [
                      {
                        value: "三全鲜食（北新泾店）",
                        address: "长宁区新渔路144号",
                      },
                      {
                        value: "Hot honey 首尔炸鸡（仙霞路）",
                        address: "上海市长宁区淞虹路661号",
                      },
                    ];
                    resolve(
                      queryString
                        ? restaurant.filter(this.createFilter(queryString))
                        : restaurant
                    );
                  }, 1000);
                });
                // 调用 callback 返回建议列表的数据
                cb(res);
              },
            },
            on: {
              select: (val) => {
                console.log(val, "autocomplete选中");
              },
            },
            scopedSlots: {
              default: ({ item }) => {
                return (
                  <div>
                    {item.value} - {item.address}
                  </div>
                );
              },
            },
          },
          {
            element: "el-select",
            field: "sex",
            title: "性别 ",
            props: {
              options: [
                {
                  label: "热门城市",
                  value: "1",
                },
                {
                  label: "女",
                  value: "2",
                },
              ],
            },

            on: {
              change: (val) => {
                console.log("change", val);
                if (val == 1) {
                  this.params.name = "北京";
                } else {
                  this.params.name = "刘亦菲";
                }
              },
            },
          },

          {
            element: "el-checkbox-group",
            field: "checkbox",
            title: "复选框 ",
            props: {
              options: [
                { value: "A", label: "复选框 A" },
                { value: "B", label: "复选框 B" },
              ],
            },
            defaultValue: [],
          },
          {
            element: "el-radio-group",
            field: "radio",
            title: "单选框 ",
            props: {
              options: [
                { value: "1", label: "单选框1" },
                { value: "2", label: "单选框2" },
              ],
            },
          },
          {
            element: "el-date-picker",
            field: "datePicker",
            title: "日期选择器 ",
            props: {
              type: "datetime",
              valueFormat: "yyyy-MM-dd HH:mm:ss",
            },
          },
        ];
      },
    },
    methods: {
      handleSubmit(val) {
        console.log(val, "点击查询");
      },
      createFilter(queryString) {
        return (restaurant) => {
          return (
            restaurant.value
              .toLowerCase()
              .indexOf(queryString.toLowerCase()) === 0
          );
        };
      },
    },
  };
</script>
